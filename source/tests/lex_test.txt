function funcao(int numero) : int {
	# Variavel "statica"
	persistent int a = 0;

	{
		#constante
		const string literal = "Literal constante";
		a = a + numero;
	}

	# condicional
	if( a == numero){
		write("Igual");
	}else{
		write("Diferente");
	}

	}

}
